var buttonEl = document.getElementById('clickyMcButton')
var buttonPowerEl = document.getElementById('buttonPower')


var click$ = Rx.Observable.fromEvent(buttonEl, 'click');

click$
  .filter(() => buttonPowerEl.checked)
  .subscribe(e => console.log('clicked'))

var subscription;

buttonPowerEl.addEventListener('click', function(e) {
  if (e.target.checked) {
    subscription = click$.subscribe(e => alert('clicked'))
  } else {
    subscription.unsubscribe()
  }
});
